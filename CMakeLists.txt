#CMakeLists.txt for Pixirad camera

SET(CMAKE_SHARED_LINKER_FLAGS "-Wl,--add-needed")

SET(pixirad_srcs src/PixiradCamera.cpp  src/PixiradDetInfoCtrlObj.cpp
	 src/PixiradInterface.cpp src/PixiradSyncCtrlObj.cpp
	 src/pixiradDetector.cpp src/PixiradReconstructionCtrlObj.cpp
	src/pixiradFpgaTools.cpp src/pixiradHelperFunctions.cpp)
	 
FILE(STRINGS "VERSION" pixirad_vers)
ADD_LIBRARY(limapixirad SHARED ${pixirad_srcs})
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")

#INCLUDES
TARGET_INCLUDE_DIRECTORIES(limapixirad PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/include")

#LIBRARIES
#NO LIBRARY

TARGET_LINK_LIBRARIES(limapixirad limacore)

SET_TARGET_PROPERTIES(limapixirad PROPERTIES VERSION ${pixirad_vers} SOVERSION "1")

INSTALL(TARGETS limapixirad LIBRARY DESTINATION lib)

IF(COMPILE_SIP)
    SET(NAME "pixirad")
    SET(INCLUDES)
    FILE(GLOB sipfiles RELATIVE "${CMAKE_CURRENT_SOURCE_DIR}/sip" "${CMAKE_CURRENT_SOURCE_DIR}/sip/*.sip")
    FOREACH(sipfile ${sipfiles})
        SET(INCLUDES 
        "${INCLUDES} 
        %Include ${sipfile}")
    ENDFOREACH()

    CONFIGURE_FILE(${CMAKE_SOURCE_DIR}/sip/limamodules.sip.in sip/limapixirad.sip)
    SET(SIP_CONCAT_PARTS 1)
    SET(SIP_INCLUDES ${SIP_INCLUDES}
        "${CMAKE_SOURCE_DIR}/third-party/Processlib/sip"
        "${CMAKE_BINARY_DIR}/sip/core"
        "${CMAKE_SOURCE_DIR}/third-party/Processlib/tasks/sip"
        "${CMAKE_SOURCE_DIR}/common/sip"
        "${CMAKE_SOURCE_DIR}/hardware/sip"
        "${CMAKE_SOURCE_DIR}/control/sip"
        "${CMAKE_SOURCE_DIR}/control/software_operation/sip"
        "${CMAKE_CURRENT_SOURCE_DIR}/sip")
    set(SIP_DISABLE_FEATURES WITH_CONFIG)
    set(SIP_CHECK_EXC ON)
    ADD_SIP_PYTHON_MODULE(limapixirad ${CMAKE_CURRENT_BINARY_DIR}/sip/limapixirad.sip)
    TARGET_INCLUDE_DIRECTORIES(python_module_limapixirad PRIVATE
                               ${PYTHON_INCLUDE_DIRS}
                               "${CMAKE_SOURCE_DIR}/sip"
                               "${CMAKE_SOURCE_DIR}/sip/core")
    TARGET_LINK_LIBRARIES(python_module_limapixirad limapixirad)

    INSTALL(FILES python/__init__.py DESTINATION "${PYTHON_SITE_PACKAGES_DIR}/Lima/Pixirad")
ENDIF()
